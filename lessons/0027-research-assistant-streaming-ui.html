Architecture

React client                  Next.js route              OpenAI
  │                                  │                        │
  │── fetch POST /api/research ─────►│ streamResearch()       │
  │    { message }                   │   stream: true         │
  │                                  │───────────────────────►│ web_search
  │◄── data: { phase: searching } ─│◄── web_search_call ────│
  │◄── data: { phase: streaming… } │◄── output_text.delta ──│
  │◄── data: { phase: done, … } ────│◄── response.completed ─│
  │    setState(snapshot)             │                        │

The client never sees raw OpenAI events — only your UI phase snapshots. Each SSE frame is a full ResearchUIState object — replace React state on every frame; the client does not interpret the API.

Research phase machine

One-turn web_search is simpler than triage's two-turn loop: 

idle → searching → streaming-answer → done
                              ↘ error
Phase User sees Triggered by
searching “Searching the web…” output_item.added with web_search_call
streaming-answer Answer text growing live output_text.delta
done Final answer + citation links response.completed — extract url_citation annotations

If the model skips search, you may never hit searching — the first frame might be streaming-answer. That is fine.

1. UI state — lib/research-state.ts

Types and createResearchState live in a client-safe module — no OpenAI imports. The client component imports from here; the stream helper imports from both this file and lib/openai.ts. 

export type Citation = { title: string; url: string };

export type ResearchPhase =
  | "idle"
  | "searching"
  | "streaming-answer"
  | "done"
  | "error";

export type ResearchUIState = {
  phase: ResearchPhase;
  answerPreview: string;
  answer: string | null;
  citations: Citation[];
  searched: boolean;
  error: string | null;
};

export function createResearchState(
  overrides: Partial<ResearchUIState> = {}
): ResearchUIState {
  return {
    phase: "idle",
    answerPreview: "",
    answer: null,
    citations: [],
    searched: false,
    error: null,
    ...overrides,
  };
}

2. Stream helper — lib/stream-research.ts

Port the shape from practice/lib/stream-triage-docs.mjs (docs path) — built-in tool, one turn, no handler. Mark server-only so the client cannot import it. 

import "server-only";

import type { Response } from "openai/resources/responses/responses";
import { openai } from "@/lib/openai";
import { createResearchState, type ResearchUIState } from "@/lib/research-state";

function extractUrlCitations(response: Response): Citation[] {
  const message = response.output.find((item) => item.type === "message");
  const textContent = message?.content?.find(
    (part) => part.type === "output_text"
  );
  return (textContent?.annotations ?? [])
    .filter((note) => note.type === "url_citation")
    .map((note) => ({ title: note.title ?? note.url, url: note.url }));
}

export async function streamResearch(
  message: string,
  onSnapshot: (state: ResearchUIState) => void
) {
  const state = createResearchState({ phase: "searching" });
  onSnapshot(state);

  const stream = await openai.responses.create({
    model: "gpt-5-mini",
    instructions:
      "You are a research assistant for solo builders. Answer clearly and concisely. Cite sources when you use web search.",
    input: message,
    tools: [{ type: "web_search", search_context_size: "low" }],
    stream: true,
  });

  for await (const event of stream) {
    if (
      event.type === "response.output_item.added" &&
      event.item.type === "web_search_call"
    ) {
      state.searched = true;
      state.phase = "searching";
      onSnapshot({ ...state });
    }

    if (event.type === "response.output_text.delta") {
      state.phase = "streaming-answer";
      state.answerPreview += event.delta;
      onSnapshot({ ...state });
    }

    if (event.type === "response.completed") {
      state.answer = event.response.output_text;
      state.citations = extractUrlCitations(event.response);
      state.searched = event.response.output.some(
        (item) => item.type === "web_search_call"
      );
      state.phase = "done";
      onSnapshot({ ...state });
    }
  }
}
Type the completed response, not create()'s return type.
responses.create returns Stream | Response depending on stream. Citation extraction runs on event.response at response.completed — always a Response — so import that type directly instead of inferring from ReturnType<typeof openai.responses.create>.
Citations at done, not during streaming.
url_citation annotations arrive on the final message. Stream answerPreview for live text; render citation links only when phase === "done" — same rule as Lesson 26.

3. SSE route — replace JSON response

Swap the blocking NextResponse.json for a ReadableStream. Set export const runtime = "nodejs" — the OpenAI SDK needs Node, not Edge. 

// app/api/research/route.ts
import { createResearchState } from "@/lib/research-state";
import { streamResearch } from "@/lib/stream-research";

export const runtime = "nodejs";

export async function POST(req: Request) {
  const body = await req.json();
  const message = body?.message;

  if (typeof message !== "string" || !message.trim()) {
    return Response.json({ error: "message is required" }, { status: 400 });
  }

  const stream = new ReadableStream({
    async start(controller) {
      const encoder = new TextEncoder();
      const send = (state: Parameters<typeof streamResearch>[1] extends (
        s: infer S
      ) => void
        ? S
        : never) =>
        controller.enqueue(
          encoder.encode(`data: ${JSON.stringify(state)}\n\n`)
        );

      try {
        await streamResearch(message.trim(), send);
      } catch (err) {
        send(
          createResearchState({
            phase: "error",
            error: err instanceof Error ? err.message : String(err),
          })
        );
      } finally {
        controller.close();
      }
    },
  });

  return new Response(stream, {
    headers: { "Content-Type": "text/event-stream" },
  });
}

Your Lesson 26 curl test no longer returns one JSON blob — that is expected. The browser client is the test surface now.

4. Client — components/ResearchChat.tsx

Keep app/page.tsx as a thin server shell; put interactivity in a client component. Install @microsoft/fetch-event-source — it handles spec-compliant SSE parsing over fetch (including POST), which native EventSource cannot do: 

npm install @microsoft/fetch-event-source
"use client";

import { fetchEventSource } from "@microsoft/fetch-event-source";
import { useRef, useState } from "react";
import { createResearchState } from "@/lib/research-state";

export function ResearchChat() {
  const [state, setState] = useState(createResearchState());
  const abortRef = useRef<AbortController | null>(null);

  async function onSubmit(e: React.FormEvent<HTMLFormElement>) {
    e.preventDefault();
    const form = e.currentTarget;
    const message = new FormData(form).get("message");
    if (typeof message !== "string" || !message.trim()) return;

    abortRef.current?.abort();
    abortRef.current = new AbortController();

    setState(createResearchState({ phase: "searching" }));

    try {
      await fetchEventSource("/api/research", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ message: message.trim() }),
        signal: abortRef.current.signal,
        async onopen(response) {
          if (!response.ok) {
            throw new Error(`Request failed (${response.status})`);
          }
        },
        onmessage(ev) {
          setState(JSON.parse(ev.data));
        },
        onerror(err) {
          throw err;
        },
      });
    } catch (err) {
      if (err instanceof Error && err.name === "AbortError") return;
      setState(
        createResearchState({
          phase: "error",
          error: err instanceof Error ? err.message : String(err),
        })
      );
    }
  }

  const busy =
    state.phase !== "idle" &&
    state.phase !== "done" &&
    state.phase !== "error";

  return (
    <>
      <form onSubmit={onSubmit} className="flex flex-col gap-4">
        <textarea
          name="message"
          className="min-h-32 w-full resize-y rounded-lg border border-zinc-300 bg-white px-4 py-3 text-base text-foreground outline-none placeholder:text-zinc-500 focus:border-zinc-400 disabled:cursor-not-allowed disabled:opacity-50 dark:border-zinc-600 dark:bg-zinc-800 dark:placeholder:text-zinc-400 dark:focus:border-zinc-500"
          placeholder="Ask a research question…"
          rows={4}
          disabled={busy}
        />
        <button
          type="submit"
          disabled={busy}
          className="self-end rounded-lg bg-zinc-900 px-5 py-2.5 text-sm font-medium text-white disabled:cursor-not-allowed disabled:opacity-50 dark:bg-zinc-100 dark:text-zinc-900"
        >
          Send
        </button>
      </form>

      {state.phase === "searching" && <p>Searching the web…</p>}
      {state.phase === "streaming-answer" && <p>{state.answerPreview}</p>}
      {state.phase === "done" && (
        <>
          <p>{state.answer}</p>
          <ul>
            {state.citations.map((c) => (
              <li key={c.url}>
                <a href={c.url} target="_blank" rel="noreferrer">
                  {c.title}
                </a>
              </li>
            ))}
          </ul>
        </>
      )}
      {state.phase === "error" && <p>{state.error}</p>}
    </>
  );
}
