Research useChat

useChat + data-research parts carrying ResearchUIState.

Message type

import type { UIMessage } from "ai";
import type { ResearchUIState } from "@/lib/research-state";

export type ResearchUIMessage = UIMessage<
  never,
  { research: ResearchUIState }
>;

Server: snapshot → data part

const RESEARCH_PART_ID = "research-1";

await streamResearch(message, (state) => {
  writer.write({
    type: "data-research",
    id: RESEARCH_PART_ID,
    data: state,
  });
}, { previousResponseId, vectorStoreId });

Client: transport + hook

import { useChat } from "@ai-sdk/react";
import { DefaultChatTransport } from "ai";

const { messages, sendMessage, status, stop, setMessages } =
  useChat<ResearchUIMessage>({
    transport: new DefaultChatTransport({
      api: "/api/research",
      headers: () => authHeaders(),
      body: () => ({ sessionId }),
    }),
  });

const busy = status === "submitted" || status === "streaming";

Read state from a message

function getResearchPart(message: ResearchUIMessage) {
  return message.parts.find((p) => p.type === "data-research");
}

const state = getResearchPart(assistantMessage)?.data;

status values

status Meaning
ready Idle — user can send
submitted Request sent, awaiting first stream chunk
streaming Response actively streaming
error Request failed — check error

What stays server-side

  • previousResponseId — session store, not client messages
  • vectorStoreId — from upload session
  • Guards, rate limits, streamResearch phase machine
