Research Brief

Structured output + web_search in one turn — citations from annotations.

Schema (lib/schemas.ts)

export const ResearchBrief = z.object({
  headline: z.string().describe("One-line takeaway"),
  summary: z.string().describe("Two or three sentence answer"),
  key_points: z.array(z.string()).describe("Three to five facts"),
  confidence: z.enum(["low", "medium", "high"]),
});

State shape

type ResearchUIState = {
  phase: ResearchPhase;
  briefPreview: string;       // raw JSON during stream
  brief: ResearchBrief | null; // Zod-validated at done
  citations: Citation[];       // from url_citation — not schema
  searched: boolean;
  error: string | null;
};
