Research Streaming UI

One-turn web_search → UI phase snapshots over SSE.

Phase machine

idle → searching → streaming-answer → done
                              ↘ error

Simpler than triage (no handler turn, no structured card yet). Same transport as fetch stream client.

State shape

type ResearchPhase =
  | "idle"
  | "searching"
  | "streaming-answer"
  | "done"
  | "error";

type ResearchUIState = {
  phase: ResearchPhase;
  answerPreview: string;
  answer: string | null;
  citations: { title: string; url: string }[];
  searched: boolean;
  error: string | null;
};

Event → state (lib/stream-research.ts)

Stream event State update
output_item.added (web_search_call) phase: "searching", searched: true
output_text.delta phase: "streaming-answer", append answerPreview
response.completed answer, citations from url_citation annotations, phase: "done"

What to render

Phase UI
searching “Searching the web…”
streaming-answer Live answer text (answerPreview)
done Final answer + citation link list
error Error message; re-enable Send

Route sketch

export const runtime = "nodejs";

export async function POST(req: Request) {
  const stream = new ReadableStream({
    async start(controller) {
      const send = (state) =>
        controller.enqueue(
          new TextEncoder().encode(`data: ${JSON.stringify(state)}\n\n`)
        );
      await streamResearch(message, send);
      controller.close();
    },
  });
  return new Response(stream, {
    headers: { "Content-Type": "text/event-stream" },
  });
}
