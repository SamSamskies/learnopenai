Fetch Stream Client

Read SSE-shaped responses with fetch — the React/Next.js default.

EventSource vs fetch

Need EventSource fetch + reader
POST body (user question) No Yes
Auth headers No Yes

Parse loop

const res = await fetch("/api/triage", {
  method: "POST",
  headers: { "Content-Type": "application/json" },
  body: JSON.stringify({ question }),
  signal: abortController.signal,
});
const reader = res.body.getReader();
const decoder = new TextDecoder();
let buffer = "";

while (true) {
  const { done, value } = await reader.read();
  if (done) break;
  buffer += decoder.decode(value, { stream: true });
  const chunks = buffer.split("\n\n");
  buffer = chunks.pop() ?? "";
  for (const chunk of chunks) {
    const line = chunk.trim();
    if (!line.startsWith("data: ")) continue;
    const state = JSON.parse(line.slice(6));
    dispatch(state); // merge into useReducer state
  }
}
